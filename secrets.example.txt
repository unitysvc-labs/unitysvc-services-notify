# secrets.example.txt — seller secrets for the notification services in this repo
#
# Every service here relays or translates a message to an upstream messaging
# provider. Each service's offering references customer_secrets that the UnitySVC
# gateway composes into the upstream URL / credentials at request time. This file
# is the catalog of those secrets: a seller exports them locally for
# `usvc_seller specs run-tests`, and uploads them to the backend seller secret
# store so gateway / integration tests resolve them via the ops_customer →
# seller-secret fallback.
#
# Setup:
#     cp secrets.example.txt secrets.txt          # then edit secrets.txt with your values
#     source secrets.txt                          # export for local run-tests
#     usvc_seller secrets upload secrets.txt      # push to the backend seller secret store
#
# secrets.txt is gitignored. If you keep an encrypted .secrets, decrypt straight
# into the uploader instead of writing plaintext to disk:
#     sops -d .secrets | usvc_seller secrets upload
#
# Sourceable shell format — each line is `export NAME="value"`.
# Per secret: REQUIRED vs OPTIONAL, and the seller-TEST vs PRODUCTION value.
# The defaults below are the seller-TEST values (the mock host) — not real
# credentials, so this example file is safe to commit. Your real values live in
# secrets.txt (gitignored) or directly in the backend secret store.

# ─────────────────────────────────────────────────────────────────────────────
# Discord — services: labs/discord-relay, labs/msg-to-discord
# ─────────────────────────────────────────────────────────────────────────────

# DISCORD_WEBHOOK_BASE — OPTIONAL (default: https://discord.com/api/webhooks)
#   seller test : https://mock.unitysvc.dev/discord/api/webhooks   ← the mock host
#   production  : leave UNSET for real Discord, or set to your own webhook proxy
export DISCORD_WEBHOOK_BASE="https://mock.unitysvc.dev/discord/api/webhooks"

# DISCORD_WEBHOOK_ID — REQUIRED
#   seller test : any non-empty value (the mock ignores it) — e.g. demo
#   production  : the <ID> from https://discord.com/api/webhooks/<ID>/<TOKEN>
export DISCORD_WEBHOOK_ID="demo"

# DISCORD_WEBHOOK_TOKEN — REQUIRED
#   seller test : any non-empty value (the mock ignores it) — e.g. demotoken
#   production  : the <TOKEN> from https://discord.com/api/webhooks/<ID>/<TOKEN>
export DISCORD_WEBHOOK_TOKEN="demotoken"
